class Solution {
public:
    int numberOfMatches(int n) {
        int sum = 0;
        while (n > 1) {
            int match = (n % 2 == 0) ? n / 2 : (n - 1) / 2;
            n = (n % 2 == 0) ? n / 2 : (n + 1) / 2;
            sum += match;
        }
        return sum;
    }
};
